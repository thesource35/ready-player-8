    File 1: web/src/lib/video/hls-sign.ts

    ```ts
    import type { SupabaseClient } from '@supabase/supabase-js'

    const BUCKET = 'videos'

    export async function signHlsManifest(
      supabase: SupabaseClient,
      hlsDir: string,
      ttlSeconds: number,
    ): Promise<{ manifestText: string } | { error: string }> {
      // 1. List all files in the hls/ subdirectory
      const { data: files, error: listErr } = await supabase.storage.from(BUCKET).list(hlsDir)
      if (listErr || !files) return { error: listErr?.message ?? 'list failed' }
      if (files.length === 0) return { error: 'empty hls directory' }

      const paths = files.map(f => `${hlsDir}/${f.name}`)

      // 2. Batch-sign every file (manifest + all segments)
      const { data: signed, error: signErr } = await supabase.storage.from(BUCKET).createSignedUrls(paths, ttlSeconds)
      if (signErr || !signed) return { error: signErr?.message ?? 'sign failed' }

      // Build filename -> signedUrl lookup
      const byName = new Map<string, string>()
      for (const s of signed) {
        if (!s.path || !s.signedUrl) continue
        const name = s.path.split('/').pop()!
        byName.set(name, s.signedUrl)
      }

      const manifestSignedUrl = byName.get('index.m3u8')
      if (!manifestSignedUrl) return { error: 'index.m3u8 missing' }

      // 3. Fetch the manifest text via its signed URL
      const res = await fetch(manifestSignedUrl)
      if (!res.ok) return { error: `fetch manifest: ${res.status}` }
      const text = await res.text()

      // 4. Rewrite segment URIs (.ts / .m4s lines) to their pre-signed absolute URLs
      const rewritten = text
        .split('\n')
        .map(line => {
          const trimmed = line.trim()
          if (!trimmed || trimmed.startsWith('#')) return line
          if (trimmed.endsWith('.ts') || trimmed.endsWith('.m4s')) {
            const signed = byName.get(trimmed)
            return signed ?? line
          }
          return line
        })
        .join('\n')

      return { manifestText: rewritten }
    }
    ```

    import { Hono } from 'hono'
    import { serve } from '@hono/node-server'
    import { timingSafeEqual } from 'node:crypto'
    import { config } from './config.js'
    import { transcodeAsset } from './transcode.js'

    const app = new Hono()

    app.get('/health', c => c.json({ ok: true }))

    app.post('/transcode', async c => {
      const provided = c.req.header('x-worker-secret') ?? ''
      const expected = config.workerSecret
      if (
        provided.length !== expected.length ||
        !timingSafeEqual(Buffer.from(provided), Buffer.from(expected))
      ) {
        return c.json({ error: 'Unauthorized' }, 401)
      }
      const body = await c.req.json().catch(() => null)
      if (!body?.asset_id || !body?.storage_path || !body?.org_id || !body?.project_id) {
        return c.json({ error: 'Missing fields' }, 400)
      }
      // Return 202 fast; process async
      setImmediate(() => {
        transcodeAsset(body).catch(err =>
          console.error('[worker] transcodeAsset top-level catch:', err)
        )
      })
      return c.json({ accepted: true }, 202)
    })

    serve({ fetch: app.fetch, port: config.port })
    console.log(`[worker] listening on :${config.port}`)

    worker/src/transcode.ts — full transcode pipeline:

    Key functions:
    - `export async function transcodeAsset({ asset_id, storage_path, org_id, project_id })`:
      1. Update row status='transcoding'
      2. Download raw.{ext} from Supabase Storage to `/tmp/${asset_id}/raw.{ext}`
      3. Probe codec with ffprobe; if not in {h264, hevc, prores} → mark status='failed' with last_error `unsupported codec: {codec}` and return
      4. Try runFfmpeg; catch failure → retry with exponential backoff (first retry after 30000ms, second after 120000ms, D-33)
      5. After success: upload index.m3u8 + segment_*.ts + poster.jpg to `videos/{org}/{project}/{asset}/hls/` (poster goes to `.../poster.jpg` outside hls/) with service-role client (bypasses RLS)
      6. Update row: status='ready', duration_s (from ffprobe), retention_expires_at = now + 30d (D-09)
      7. Emit analytics video_transcode_succeeded (wire in plan 22-11; here console.log a structured line the analytics pipeline can consume)
      8. Clean /tmp/${asset_id}
      9. On final failure after retries: status='failed', last_error, emit video_transcode_failed
    - `async function runFfmpeg(inputPath, outDir)`: spawns `ffmpeg -i {input} -y -vf scale=1280:720 -c:v libx264 -preset veryfast -profile:v main -b:v 2500k -maxrate 2675k -bufsize 5000k -c:a aac -b:a 128k -ac 2 -f hls -hls_time 6 -hls_list_size 0 -hls_segment_filename '{outDir}/segment_%03d.ts' -hls_playlist_type vod {outDir}/index.m3u8`. Use `child_process.spawn` and await exit code 0. On non-zero exit, throw.
    - `async function extractPoster(inputPath, outPath)`: spawns `ffmpeg -i {input} -ss 00:00:03 -frames:v 1 -q:v 3 {outPath} -y` — don't fail the whole transcode if poster fails (best-effort, log a warning).
    - `async function probeCodec(inputPath): Promise<{ codec: string, duration: number }>`: runs `ffprobe -v error -select_streams v:0 -show_entries stream=codec_name:format=duration -of json {input}`, parses JSON, returns { codec, duration }.
