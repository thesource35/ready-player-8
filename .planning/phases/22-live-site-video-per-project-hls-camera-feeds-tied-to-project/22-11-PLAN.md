      | ({ event: 'portal_video_view'; asset_id: string; portal_link_id: string } & BasePayload)

    export function emitVideoEvent(payload: VideoEventPayload): void {

    // Phase 22 D-40 — iOS video analytics emission helper.
    import Foundation

    enum VideoAnalytics {
        static func track(_ event: String, properties: [String: Any]) {
            // Sanitize
            var safe = properties
            safe.removeValue(forKey: "stream_key")
            safe.removeValue(forKey: "signed_url")
            AnalyticsEngine.shared.track(event, properties: safe)
        }

        static func uploadStarted(assetId: UUID, fileSizeBytes: Int, container: String, clientDurationEstimate: Double?, projectId: UUID, orgId: UUID, userId: UUID?) {
            var props: [String: Any] = ["asset_id": assetId.uuidString, "file_size_bytes": fileSizeBytes, "container": container, "project_id": projectId.uuidString, "org_id": orgId.uuidString]
            if let clientDurationEstimate { props["client_duration_estimate"] = clientDurationEstimate }
            if let userId { props["user_id"] = userId.uuidString }
            track("video_upload_started", properties: props)
        }

        static func uploadFailed(assetId: UUID, errorCode: String, bytesSent: Int, projectId: UUID, orgId: UUID, userId: UUID?) {
            var props: [String: Any] = ["asset_id": assetId.uuidString, "error_code": errorCode, "bytes_sent": bytesSent, "project_id": projectId.uuidString, "org_id": orgId.uuidString]
            if let userId { props["user_id"] = userId.uuidString }
            track("video_upload_failed", properties: props)
        }

        static func playbackStarted(assetId: UUID?, sourceId: UUID?, kind: String, qualityRendition: String?, isCellular: Bool?, projectId: UUID, orgId: UUID, userId: UUID?) {
            var props: [String: Any] = ["kind": kind, "project_id": projectId.uuidString, "org_id": orgId.uuidString]

            if let sourceId { props["source_id"] = sourceId.uuidString }

            if let isCellular { props["is_cellular"] = isCellular }

            track("video_playback_started", properties: props)
        }
    }
