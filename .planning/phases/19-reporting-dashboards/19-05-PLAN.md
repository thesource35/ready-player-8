```swift
import SwiftUI
import Charts

struct BudgetPieChartView: View {
    let spent: Double
    let remaining: Double
    var mini: Bool = false
    
    var body: some View {
        let data = [
            ChartSegment(name: "Spent", value: spent, color: Theme.accent),
            ChartSegment(name: "Remaining", value: remaining, color: Theme.cyan),
        ]
        let pct = spent + remaining > 0 ? Int((spent / (spent + remaining)) * 100) : 0
        
        Chart(data, id: \.name) { segment in
            SectorMark(
                angle: .value(segment.name, segment.value),
                innerRadius: .ratio(mini ? 0.55 : 0.65),
                angularInset: 1.5
            )
            .foregroundStyle(segment.color)
            .cornerRadius(4)
        }
        .frame(height: mini ? 120 : 200)
        .overlay {
            // Center percentage text
            Text("\(pct)%")
                .font(.system(size: mini ? 16 : 24, weight: .heavy))
                .foregroundColor(Theme.text)
        }
        // Inline labels below chart (per D-23)
        HStack(spacing: 12) {
            ForEach(data, id: \.name) { segment in
                HStack(spacing: 4) {
                    Circle().fill(segment.color).frame(width: 8, height: 8)
                    Text(segment.name).font(.system(size: 8, weight: .heavy)).foregroundColor(Theme.muted)
                }
            }
        }
    }
}

struct ChartSegment {
    let name: String
    let value: Double
    let color: Color
}
```

```swift
import UIKit
import SwiftUI

struct ReportPDFGenerator {
    static func generatePDF(
        projectName: String,
        clientName: String,
        health: HealthScoreData,
        sections: ProjectReportData
    ) -> Data {
        let pageSize = CGSize(width: 612, height: 792) // US Letter (8.5x11 inches at 72 DPI)
        let margins = UIEdgeInsets(top: 54, left: 54, bottom: 54, right: 54) // 0.75 inches
        let contentRect = CGRect(
            x: margins.left, y: margins.top,
            width: pageSize.width - margins.left - margins.right,
            height: pageSize.height - margins.top - margins.bottom
        )
        
        let renderer = UIGraphicsPDFRenderer(bounds: CGRect(origin: .zero, size: pageSize))
        
        let data = renderer.pdfData { context in
            // Page 1: Header + Budget + Schedule
            context.beginPage()
            drawHeader(in: context, rect: contentRect, projectName: projectName, clientName: clientName)
            // ... draw budget section, schedule section
            
            // Additional pages as needed
            // drawFooter on each page: page number centered, "Generated by ConstructionOS" left
        }
        
        return data
    }
    
    private static func drawHeader(in context: UIGraphicsPDFRendererContext, rect: CGRect, projectName: String, clientName: String) {
        // Project name: 18pt bold at top center
        let titleAttrs: [NSAttributedString.Key: Any] = [
            .font: UIFont.systemFont(ofSize: 18, weight: .bold),
            .foregroundColor: UIColor(red: 0.1, green: 0.1, blue: 0.1, alpha: 1)
        ]
        let title = NSAttributedString(string: projectName, attributes: titleAttrs)
        title.draw(at: CGPoint(x: rect.midX - title.size().width / 2, y: rect.minY))
        
        // Date: 10pt right-aligned
        let dateFormatter = DateFormatter()
        dateFormatter.dateFormat = "MMMM d, yyyy 'at' h:mm a"
        let dateStr = "Generated \(dateFormatter.string(from: Date()))"
        let dateAttrs: [NSAttributedString.Key: Any] = [
            .font: UIFont.systemFont(ofSize: 10),
            .foregroundColor: UIColor.gray
        ]
        let date = NSAttributedString(string: dateStr, attributes: dateAttrs)
        date.draw(at: CGPoint(x: rect.maxX - date.size().width, y: rect.minY))
    }
    
    private static func drawFooter(in context: UIGraphicsPDFRendererContext, pageSize: CGSize, pageNumber: Int, totalPages: Int) {
        let footerY = pageSize.height - 36 // 0.5 inch from bottom
        // "Generated by ConstructionOS" left
        let leftAttrs: [NSAttributedString.Key: Any] = [
            .font: UIFont.systemFont(ofSize: 8), .foregroundColor: UIColor.gray
        ]
        NSAttributedString(string: "Generated by ConstructionOS", attributes: leftAttrs)
            .draw(at: CGPoint(x: 54, y: footerY))
        // Page number centered
        let pageStr = "Page \(pageNumber) of \(totalPages)"
        let pageAttrs: [NSAttributedString.Key: Any] = [
            .font: UIFont.systemFont(ofSize: 9), .foregroundColor: UIColor.darkGray
        ]
        let pageText = NSAttributedString(string: pageStr, attributes: pageAttrs)
        pageText.draw(at: CGPoint(x: pageSize.width / 2 - pageText.size().width / 2, y: footerY))
    }
    
    // Share PDF via UIActivityViewController
    static func sharePDF(data: Data, filename: String, from viewController: UIViewController? = nil) {
        let tempURL = FileManager.default.temporaryDirectory.appendingPathComponent(filename)
        try? data.write(to: tempURL)
        let activityVC = UIActivityViewController(activityItems: [tempURL], applicationActivities: nil)
        // Present from root or provided VC
        if let vc = viewController ?? UIApplication.shared.connectedScenes
            .compactMap({ $0 as? UIWindowScene }).first?.windows.first?.rootViewController {
            vc.present(activityVC, animated: true)
        }
    }
}
```
