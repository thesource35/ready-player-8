CRITICAL: Helper to generate REAL HMAC-SHA256 signatures using crypto.subtle (same algorithm as route):
```typescript
async function generateSignature(body: string, secret: string, timestampOverride?: number): Promise<string> {
  const timestamp = timestampOverride ?? Math.floor(Date.now() / 1000);
  const encoder = new TextEncoder();
  const key = await crypto.subtle.importKey(
    "raw", encoder.encode(secret),
    { name: "HMAC", hash: "SHA-256" }, false, ["sign"]
  );
  const payload = `${timestamp}.${body}`;
  const sig = await crypto.subtle.sign("HMAC", key, encoder.encode(payload));
  const hex = Array.from(new Uint8Array(sig)).map(b => b.toString(16).padStart(2, "0")).join("");
  return `ts=${timestamp};h1=${hex}`;
}
```
