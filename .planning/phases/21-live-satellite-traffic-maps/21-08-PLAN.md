**Defect 6.1 — Camera/geolocation race (L263-300):** resolve by applying the rule "if a saved camera exists, skip geolocation entirely. If no saved camera, geolocation may fly to user position."

Inside `map.on("load", () => { ... })`:

```tsx
map.on("load", () => {
  setMapLoaded(true);

  const savedCamera = localStorage.getItem(MAP_STORAGE_KEYS.cameraPrefix + "default");
  let cameraRestored = false;
  if (savedCamera) {
    try {
      const { center, zoom } = JSON.parse(savedCamera);
      map.setCenter(center);
      map.setZoom(zoom);
      cameraRestored = true;
    } catch {
      // ignore invalid saved camera
    }
  }

  // Save camera position on move (existing behavior)
  map.on("moveend", () => {
    const center = [map.getCenter().lng, map.getCenter().lat];
    localStorage.setItem(
      MAP_STORAGE_KEYS.cameraPrefix + "default",
      JSON.stringify({ center, zoom: map.getZoom() })
    );
  });

  rebuildSiteMarkers(map);
  loadMapData();

  // Apply persisted overlay state on first paint (closes defects 6.2 and 6.3).
  // Without this, effects at L311 and L333 bail on the initial render because mapRef.current
  // is assigned AFTER loadMap() resolves; their deps don't re-trigger once the ref populates.
  if (activeOverlays.has("TRAFFIC")) addTrafficLayer(map);
  if (!activeOverlays.has("SATELLITE")) {
    // Only switch if default style (satellite-streets-v12) doesn't match persisted state.
    map.setStyle("mapbox://styles/mapbox/dark-v11");
    setMapStyle("dark");
    map.once("style.load", () => {
      rebuildSiteMarkers(map);
      rebuildEquipmentMarkers(map);
      rebuildPhotoMarkers(map);
      // TRAFFIC re-added after style.load if persisted on
      if (activeOverlays.has("TRAFFIC")) addTrafficLayer(map);
    });
  }

  // Geolocation ONLY when no saved camera — otherwise the flyTo clobbers the restored camera.
  if (!cameraRestored && navigator.geolocation) {
    navigator.geolocation.getCurrentPosition(
      (pos) => {
        map.flyTo({ center: [pos.coords.longitude, pos.coords.latitude], zoom: 12 });
      },
      () => { /* keep default */ },
      { timeout: 3000 }
    );
  }
});
```
