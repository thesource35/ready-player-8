Create `web/src/middleware.ts` as a proper Next.js middleware file. This replaces the functionality of proxy.ts.

**IMPORTANT:** Next.js middleware must be at `web/src/middleware.ts` (inside the `src` directory since the project uses `src/` layout). The middleware function must be the default export OR a named export called `middleware`. It must also export a `config` object with a `matcher` array.

The middleware must:
1. Use `@supabase/ssr` `createServerClient` to create a Supabase client that can read/write cookies from the request/response
2. Call `supabase.auth.getUser()` to validate the session (this refreshes the token if needed and is the recommended approach per Supabase docs)
3. If no valid session and route is protected, redirect to `/login?redirect={pathname}`
4. If Supabase env vars not configured, allow all routes (demo mode per AUTH-06 equivalent for web)
5. Pass through public routes, API routes, static files, and auth routes

```typescript
import { NextResponse } from "next/server";
import type { NextRequest } from "next/server";
import { createServerClient } from "@supabase/ssr";

const publicRoutes = [
  "/",
  "/login",
  "/pricing",
  "/terms",
  "/privacy",
  "/checkout",
  "/verify",
  "/profile",
  "/auth/callback",
  "/auth/signout",
];

export async function middleware(request: NextRequest) {
  const { pathname } = request.nextUrl;

  // Allow public routes
  if (publicRoutes.some((route) => pathname === route)) {
    return NextResponse.next();
  }

  // Allow API routes (they handle their own auth)
  if (pathname.startsWith("/api/")) {
    return NextResponse.next();
  }

  // Allow static files and Next.js internals
  if (pathname.startsWith("/_next/") || pathname.includes(".")) {
    return NextResponse.next();
  }

  // If Supabase is not configured, allow all routes (demo mode)
  const url = process.env.NEXT_PUBLIC_SUPABASE_URL;
  const key = process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY;
  if (!url || !key) {
    return NextResponse.next();
  }

  // Create Supabase client with cookie management for middleware
  let response = NextResponse.next({
    request: { headers: request.headers },
  });

  const supabase = createServerClient(url, key, {
    cookies: {
      getAll() {
        return request.cookies.getAll();
      },
      setAll(cookiesToSet) {
        cookiesToSet.forEach(({ name, value }) =>
          request.cookies.set(name, value)
        );
        response = NextResponse.next({
          request: { headers: request.headers },
        });
        cookiesToSet.forEach(({ name, value, options }) =>
          response.cookies.set(name, value, options)
        );
      },
    },
  });

  // Validate session by calling getUser (refreshes token if needed)
  const {
    data: { user },
  } = await supabase.auth.getUser();

  if (!user) {
    const loginUrl = new URL("/login", request.url);
    loginUrl.searchParams.set("redirect", pathname);
    return NextResponse.redirect(loginUrl);
  }

  return response;
}

export const config = {
  matcher: [
    /*
     * Match all request paths except:
     * - _next/static (static files)
     * - _next/image (image optimization files)
     * - favicon.ico, sitemap.xml, robots.txt (metadata files)
     */
    "/((?!_next/static|_next/image|favicon.ico|sitemap.xml|robots.txt).*)",
  ],
};
```

export async function POST(req: Request) {
  // Rate limit: 5 lead submissions per minute per IP (AUTH-11)

    || req.headers.get("x-real-ip")
    || "unknown";

    );
